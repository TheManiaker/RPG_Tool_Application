create database game character set utf8 collate utf8_unicode_ci;
use game;
create table account(id int auto_increment, login varchar(20) not null, pass varchar(255) not null, primary key (id));
create table game_char(id int auto_increment, name varchar(20) not null, char_file varchar(255), primary key (id));
create table char_acc(id_acc int not null, id_char int not null, foreign key (id_acc) references account (id) on delete cascade, foreign key (id_char) references game_char (id) on delete cascade);
create table room(id int auto_increment, name varchar(20) not null, max_users int not null, create_date date, room_file varchar(255), primary key (id));
